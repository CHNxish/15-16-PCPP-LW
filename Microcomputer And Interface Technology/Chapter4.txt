分支结构
x，y中包含元素x1，x2
1)当x与y不同奇同偶时，将奇数存入x，将偶数存入y
2)当x与y同奇时，两数分别加1
3)当x与y同偶时，两数不动

Data    segment
X       db      x1
Y       db      x2
data    ends

Stack   segment stack

Stack   ends

Code    segmnt
        assume  cs:Code,ds:Data
        assume  ss:Stack
start:  mov     ax,Data
        mov     ds,ax
        mov     al,X
        mov     bl,Y
        xor     al,bl
        test    al,01h
        jz      totle
        test    bl,01h
        jmp     exit
        xchg    bl,X
        mov     Y,bl
        jmp     exit
totle:  test    bl,01h
        jz      exit
        inc     X
        inc     Y
exit:   hlt
Code    ends
        end     start

循环结构
以sta的为首地址存储区有count个字节数，统计零数据的个数送入zero单元

Data    segment
sta     db      01h,0,4bh,0
count   db      $ - sta
zero    db      ?
Data    ends

Stack   segment stack

Stack   ends

Code    segment
        assume  cs:Code,ds:Data
        assume  ss:Stack
start:  mov     ax,Data
        mov     ds,ax
        mov     cx,word ptr count
        lea     si,sta
        mov     dl,0
lop:    mov     al,[si]
        cmp     al,0
        jz      last
        inc     dl
last:   inc     si
        loop    lop
        mov     count,dl
        mov     ah,02h
        int     21h
        
        mov     ah,4ch
        int     21h
Code    s
        end     start
